91  std::vector<std::string> Wrap(const std::string &in, const std::string::size_type width, std::string::size_type firstlinewidth = 0)
92  {
93  // Preserve existing line breaks
94  const auto newlineloc = in.find('\n');
95  if (newlineloc != in.npos)
96  {
97  auto first = Wrap(std::string(in, 0, newlineloc), width);
98  auto second = Wrap(std::string(in, newlineloc + 1), width);
99  first.insert(
100  std::end(first),
101  std::make_move_iterator(std::begin(second)),
102  std::make_move_iterator(std::end(second)));
103  return first;
104  }
105  if (firstlinewidth == 0)
106  {
107  firstlinewidth = width;
108  }
109  auto currentwidth = firstlinewidth;
110 
111  std::istringstream stream(in);
112  std::vector<std::string> output;
113  std::ostringstream line;
114  std::string::size_type linesize = 0;
115  while (stream)
116  {
117  std::string item;
118  stream >> item;
119  auto itemsize = Glyphs(item);
120  if ((linesize + 1 + itemsize) > currentwidth)
121  {
122  if (linesize > 0)
123  {
124  output.push_back(line.str());
125  line.str(std::string());
126  linesize = 0;
127  currentwidth = width;
128  }
129  }
130  if (itemsize > 0)
131  {
132  if (linesize)
133  {
134  ++linesize;
135  line << " ";
136  }
137  line << item;
138  linesize += itemsize;
139  }
140  }
141  if (linesize > 0)
142  {
143  output.push_back(line.str());
144  }
145  return output;
146  }

1154  // Reset all Matched statuses and errors
1155  Reset();

1157 
1158  // Check all arg chunks
1159  for (auto it = begin; it != end; ++it)
1160  {
1161  const auto &chunk = *it;
1162 
1163  if (!terminated and chunk == terminator)
1164  {
1165  terminated = true;
1166  // If a long arg was found
1167  } else if (!terminated && chunk.find(longprefix) == 0 && chunk.size() > longprefix.size())
1168  {
1169  const auto argchunk = chunk.substr(longprefix.size());
1170  // Try to separate it, in case of a separator:
1171  const auto separator = longseparator.empty() ? argchunk.npos : argchunk.find(longseparator);
1172  // If the separator is in the argument, separate it.
1173  const auto arg = (separator != argchunk.npos ?
1174  std::string(argchunk, 0, separator)
1175  : argchunk);
1176 
1177  if (auto base = Match(arg))
1178  {
1179  if (auto argbase = dynamic_cast<ValueFlagBase *>(base))
1180  {
1181  if (separator != argchunk.npos)
1182  {
1183  if (allowJoinedLongValue)
1184  {
1185  argbase->ParseValue(argchunk.substr(separator + longseparator.size()));
1186  } else
1187  {
1188 #ifdef ARGS_NOEXCEPT
1189  error = Error::Parse;
1190  return it;
1191 #else
1192  std::ostringstream problem;
1193  problem << "Flag '" << arg << "' was passed a joined argument, but these are disallowed";
1194  throw ParseError(problem.str());
1195 #endif
1196  }
1197  } else
1198  {
1199  ++it;
1200  if (it == end)

1202 #ifdef ARGS_NOEXCEPT
1203  error = Error::Parse;
1204  return it;
1205 #else

1207  problem << "Flag '" << arg << "' requires an argument but received none";
1208  throw ParseError(problem.str());
1209 #endif
1210  }

1212  if (allowSeparateLongValue)

1217 #ifdef ARGS_NOEXCEPT

1219  return it;

1222  problem << "Flag '" << arg << "' was passed a separate argument, but these are disallowed";
1223  throw ParseError(problem.str());
1224 #endif
1225  }
1226  }
1227  } else if (separator != argchunk.npos)
1228  {
1229 #ifdef ARGS_NOEXCEPT

1231  return it;
1232 #else

1236 #endif

1250  problem << "Flag could not be matched: " << arg;
1251  throw ParseError(problem.str());
1252 #endif
1253  }
1254  // Check short args
1255  } else if (!terminated && chunk.find(shortprefix) == 0 && chunk.size() > shortprefix.size())
1256  {
1257  const auto argchunk = chunk.substr(shortprefix.size());
1258  for (auto argit = std::begin(argchunk); argit != std::end(argchunk); ++argit)
1259  {
1260  const auto arg = *argit;
1261 
1262  if (auto base = Match(arg))
1263  {
1264  if (auto argbase = dynamic_cast<ValueFlagBase *>(base))
1265  {

1280  throw ParseError(problem.str());

1282  }
1283  } else

1285  ++it;

1287  {
1288 #ifdef ARGS_NOEXCEPT
1289  error = Error::Parse;
1290  return it;
1291 #else
1292  std::ostringstream problem;
1293  problem << "Flag '" << arg << "' requires an argument but received none";
1294  throw ParseError(problem.str());
1295 #endif
1296  }
1297 
1298  if (allowSeparateShortValue)
1299  {
1300  argbase->ParseValue(*it);
1301  } else

1303 #ifdef ARGS_NOEXCEPT
1304  error = Error::Parse;
1305  return it;

1308  problem << "Flag '" << arg << "' was passed a separate argument, but these are disallowed";
1309  throw ParseError(problem.str());
1310 #endif
1311  }
1312  }
1313  // Because this argchunk is done regardless
1314  break;
1315  }
1316 
1317  if (base->KickOut())
1318  {
1319  return ++it;
1320  }
1321  } else

1323 #ifdef ARGS_NOEXCEPT

1332  }
1333  } else

1361 #else

1363  problem << "Group validation failed somewhere!";
1364  throw ValidationError(problem.str());
1365 #endif
1366  }
1367  return end;
